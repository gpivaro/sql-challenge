Employees
-
emp_no PK int
emp_title_id VARCHAR(5)
birth_date DATETIME
first_name VARCHAR(30)
last_name VARCHAR(30)
sex VARCHAR(1)
hire_date DATETIME


Departments
-
dept_no PK VARCHAR(5)
dept_name VARCHAR(50)


DepartmentsEmployess
-
emp_no int FK - Employees.emp_no
dept_no VARCHAR(5) FK >- Departments.dept_no


DepartmentsManager
-
dept_no VARCHAR(5) FK >- Departments.dept_no
emp_no int FK - Employees.emp_no


Salaries
-
emp_no int FK - Employees.emp_no
salary int


Titles
-
title_id VARCHAR(5) FK -< Employees.emp_title_id
title VARCHAR(30)
